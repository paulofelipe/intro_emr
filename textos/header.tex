% Alterando estilo do título
\makeatletter
\def\@maketitle{%
  \newpage
  \thispagestyle{empty}
  \null
  \vskip 6em%
  \begin{center}%
  \let \footnote \thanks
  \begin{flushleft}
  \vskip 10em%
    {\Huge \bfseries \sffamily \MakeUppercase \@title \par}%
    \vskip 1.5em%
    {\large
      \lineskip .2em%
      \bfseries \sffamily \@author
      \par}%
  \end{flushleft}
    \vskip 1em%
    %{\large \@date}%
  \end{center}%
  \par
  \vskip 1.5em
  \newpage}
\makeatother
